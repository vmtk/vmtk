endforeach (SCRIPT_FILE)

install(PROGRAMS ${CONTRIB_STRIPPED_SCRIPTS_SRCS}
  DESTINATION ${VMTK_CONTRIB_SCRIPTS_INSTALL_BIN_DIR}
  COMPONENT RuntimeExecutables)

endif (NOT WIN32 AND NOT VMTK_MINIMAL_INSTALL)
